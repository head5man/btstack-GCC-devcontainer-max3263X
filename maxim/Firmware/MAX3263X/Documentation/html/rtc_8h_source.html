357 __STATIC_INLINE void RTC_SetSnoozeMode(rtc_snooze_t mode)
358 {
359  uint32_t ctrl;
360  // Get the control register and mask off the non-snooze bits
361  ctrl = (MXC_RTCTMR->ctrl & ~(MXC_F_RTC_CTRL_SNOOZE_ENABLE));
362  // set the requested snooze mode bits and save the settings
363  MXC_RTCTMR->ctrl = (ctrl | (mode << MXC_F_RTC_CTRL_SNOOZE_ENABLE_POS));
364 
365  //wait for pending actions to complete
366  while(MXC_RTCTMR->ctrl & MXC_F_RTC_CTRL_PENDING);
367 }
