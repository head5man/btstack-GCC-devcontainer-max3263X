117 {

119 
120  MXC_PTG->enable |= (1 << ptIndex);

284 __STATIC_INLINE void PT_SetRestart(mxc_pt_regs_t *ptToRestart, mxc_pt_regs_t *ptStop, uint8_t restartIndex)
285 {
286  int ptStopIndex = MXC_PT_GET_IDX(ptStop);
287 
288  MXC_ASSERT(ptStopIndex >= 0);
289 
290  if(restartIndex) {
291  ptToRestart->restart |= (ptStopIndex << MXC_F_PT_RESTART_PT_Y_SELECT_POS) |
292  MXC_F_PT_RESTART_ON_PT_Y_LOOP_EXIT;
293  } else {
294  ptToRestart->restart |= (ptStopIndex << MXC_F_PT_RESTART_PT_X_SELECT_POS) |
295  MXC_F_PT_RESTART_ON_PT_X_LOOP_EXIT;
296  }
297 }
