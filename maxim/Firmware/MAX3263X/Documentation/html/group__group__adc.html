Detailed Description

ADC Overview and Usage

The ADC supports both synchronous and asynchronous access.Usage Examples

// Synchronous example
uint16_t adc_val[2]; // ADC conversion result storage
unsigned int overflow[2]; // ADC overflow storage
/* Initialize ADC */
ADC_Init();
/* Set up LIMIT0 to monitor high and low trip points on ADC Channel 1 */
ADC_SetLimit(ADC_LIMIT_0, ADC_CH_1, 1, 0x25, 1, 0x300);
/* Convert channel 0 */
ADC_StartConvert(ADC_CH_0, 0, 1);
/* Read the result into adc_val[0], store overflow flag into overflow[0] */
overflow[0] = (ADC_GetData(&adc_val[0]) == E_OVERFLOW ? 1 : 0);
/* Convert channel 1 */
ADC_StartConvert(ADC_CH_1, 0, 1);
/* Read the result into adc_val[1], store overflow flag into overflow[1] */
overflow[1] = (ADC_GetData(&adc_val[1]) == E_OVERFLOW ? 1 : 0);
/* Determine if programmable limits on AIN1 were exceeded */
if (ADC_GetFlags() & MXC_F_ADC_INTR_ADC_LO_LIMIT_IF) {
/* Low Limit exceeded */
/* ..... */
}
else if (ADC_GetFlags() & MXC_F_ADC_INTR_ADC_HI_LIMIT_IF) {
/* High Limit exceeded */
/* ..... */
}
/* Clear the ADC flags */
ADC_ClearFlags(MXC_F_ADC_INTR_ADC_LO_LIMIT_IF | MXC_F_ADC_INTR_ADC_HI_LIMIT_IF);
/***** Global variable that indicates when the adc conversion is finished *****/
volatile unsigned int adc_done = 0;
/*
* ADC IRQ Handler. This is serviced when the ADC conversion is complete.
*/
void AFE_IRQHandler(void)
{
/* Clear the ADC interrupt flags */
ADC_ClearFlags(MXC_F_ADC_INTR_ADC_DONE_IF);
/* Signal application that conversion is complete. */
adc_done = 1;
return;
}
/* ADC Asynchronous Example Snippet */
int main(void)
{
uint16_t adc_val[2];
unsigned int overflow[2];
NHD12832_ShowString((uint8_t*)"MAX3263x ADC Demo (I)", 0, 4);
/* Initialize ADC */
ADC_Init();
/* Set up LIMIT0 to monitor high and low trip points */
ADC_SetLimit(ADC_LIMIT_0, ADC_CH_1, 1, 0x25, 1, 0x300);
/* Turn on the ADC IRQ */
NVIC_EnableIRQ(AFE_IRQn);
/* mark adc conversion as not complete. */
adc_done = 0;
/* Convert channel 0 */
ADC_StartConvert(ADC_CH_0, 0, 1);
/* wait for our IRQ handler to signal ADC conversion complete */
while (!adc_done);
/* Get the converted data and it's overflow status. */
overflow[0] = (ADC_GetData(&adc_val[0]) == E_OVERFLOW ? 1 : 0);
/* Check data for overflow .... */
}
