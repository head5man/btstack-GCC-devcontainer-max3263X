154 __STATIC_INLINE void I2CS_Write(mxc_i2cs_regs_t *i2cs, uint8_t addr, uint8_t data)
155 {
156  // Make sure we don't overflow
157  MXC_ASSERT(addr < MXC_CFG_I2CS_BUFFER_SIZE);
158  i2cs->data_byte[addr] = ((i2cs->data_byte[addr] & ~MXC_F_I2CS_DATA_BYTE_DATA_FIELD) |
159  (data << MXC_F_I2CS_DATA_BYTE_DATA_FIELD_POS));
160 }

172 __STATIC_INLINE uint8_t I2CS_Read(mxc_i2cs_regs_t *i2cs, uint8_t addr)
173 {
174  // Make sure we don't overflow
175  MXC_ASSERT(addr < MXC_CFG_I2CS_BUFFER_SIZE);
176  return ((i2cs->data_byte[addr] & MXC_F_I2CS_DATA_BYTE_DATA_FIELD) >>
177  MXC_F_I2CS_DATA_BYTE_DATA_FIELD_POS);
178 }

190 __STATIC_INLINE void I2CS_SetRO(mxc_i2cs_regs_t *i2cs, uint8_t addr)
